import { AppDataSource } from "../../data-source";
import { {name} } from "../Entity/{name}";
import { Repository } from "typeorm";

export class {name}Repository extends Repository<{name}> {
    constructor() {
        super({name}, AppDataSource.manager);
    }

    /**
     * Exemple de méthode personnalisée style Symfony
     * @param id 
     */
    async findOneById(id: number): Promise<{name} | null> {
        return this.findOneBy({ id: id as any });
    }
}

export const {name}Repo = new {name}Repository();
